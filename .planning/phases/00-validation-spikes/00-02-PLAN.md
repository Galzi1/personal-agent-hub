**Spend tracking setup**

```python
import time

spend_usd = 0.0
SPEND_CAP = 18.0  # Abort at $18 to leave buffer before $20 hard cap (D-22)

def call_openrouter(model: str, messages: list, task_label: str) -> dict:
    global spend_usd
    if spend_usd >= SPEND_CAP:
        raise RuntimeError(f"Spend cap ${SPEND_CAP} reached - aborting spike. Total spent: ${spend_usd:.4f}")

    start = time.time()
    resp = httpx.post(
        "https://openrouter.ai/api/v1/chat/completions",
        headers=headers,
        json={"model": model, "messages": messages},
        timeout=120,
    )
    resp.raise_for_status()
    latency = time.time() - start
    data = resp.json()

    # OpenRouter returns usage in response body
    usage = data.get("usage", {})
    # Cost is sometimes in response headers or usage object
    cost = usage.get("cost", 0.0)  # USD; may be 0 if not returned
    spend_usd += cost

    print(f"[{task_label}] model={model} latency={latency:.1f}s cost=${cost:.4f} total=${spend_usd:.4f}")
    return {"output": data["choices"][0]["message"]["content"], "latency": latency, "cost": cost}
```
